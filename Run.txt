# --- SYSTEM PACKAGES ---
sudo apt update
sudo apt upgrade -y

# GStreamer and plugins for RTSP/H.264
sudo apt install -y \
    gstreamer1.0-tools \
    gstreamer1.0-plugins-base \
    gstreamer1.0-plugins-good \
    gstreamer1.0-plugins-bad \
    gstreamer1.0-plugins-ugly \
    gstreamer1.0-libav

# OpenCV with GStreamer support (recommended via apt)
sudo apt install -y python3-opencv

# Python and pip
sudo apt install -y python3-pip

# Kernel headers, DKMS, build-essential (for Hailo PCIe driver)
sudo apt install -y dkms build-essential raspberrypi-kernel-headers

# --- PYTHON PACKAGES ---
# If you need extra Python packages (e.g., numpy)
pip3 install numpy
